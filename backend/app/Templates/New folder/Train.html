from sklearn.metrics import accuracy_score , precision_recall_fscore_support, classification_report

def result(y_pred , y_test):
    accuracy = accuracy_score(y_test, y_pred) * 100
    precision , recall, f1_score, support = precision_recall_fscore_support(y_test, y_pred, average="weighted")
    
    res = {
        "Accuracy": accuracy,
        "Precision" : precision,
        "Recall" : recall,
        "F1-Score" : f1_score,
        "Support" : support
    }
    print(classification_report(y_test, y_pred ,target_names=['benign', 'defacement','phishing','malware']))    
    return res

def train_and_evaluate_models(models, X_train, Y_train, X_test, Y_test, save_dir='saved_models'):
    
    # Train and evaluate each model
    for model_name, model in models.items():
        print(colored(f"Training {model_name}...", 'red', attrs=['bold']))

        # Fit model to training data
        model.fit(X_train, Y_train)
        
        # Make predictions on training and testing data
        y_train_pred = model.predict(X_train)
        y_test_pred = model.predict(X_test)

        # Calculate performance metrics for Train Data
        train_metrics = {
            'accuracy': accuracy_score(Y_train, y_train_pred),
            'precision': precision_score(Y_train, y_train_pred, average='weighted'),
            'recall': recall_score(Y_train, y_train_pred, average='weighted'),
            'f1': f1_score(Y_train, y_train_pred, average='weighted')
        }

        # Calculate performance metrics for Test Data
        test_metrics = {
            'accuracy': accuracy_score(Y_test, y_test_pred),
            'precision': precision_score(Y_test, y_test_pred, average='weighted'),
            'recall': recall_score(Y_test, y_test_pred, average='weighted'),
            'f1': f1_score(Y_test, y_test_pred, average='weighted')
        }

        # Generate confusion matrix
        cm = confusion_matrix(Y_test, y_test_pred)
        plt.figure(figsize=(8, 6))
        sns.heatmap(cm, annot=True, cmap='Blues', fmt='g', cbar=False)
        plt.xlabel('Predicted Labels')
        plt.ylabel('True Labels')
        plt.title(f'Confusion Matrix - {model_name}')
        plt.show()

        print("=" * 50)
        print(colored(f"{model_name} Classifier: Training Results", 'green', attrs=['bold']))
        for metric, value in train_metrics.items():
            print(f"Training {metric.capitalize()}: {value:.4f}")

        print("=" * 50)
        print(colored(f"{model_name} Classifier: Testing Results", 'green', attrs=['bold']))
        for metric, value in test_metrics.items():
            print(f"Testing {metric.capitalize()}: {value:.4f}")

        print("=" * 50)
        print("\n" * 2)  # Adding space for readability

from sklearn.model_selection import GridSearchCV, StratifiedKFold

import joblib 

def tune_and_evaluate_models(models, param_grid, X_train, Y_train, X_test, Y_test):
    cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
    scoring_metric = make_scorer(accuracy_score)

    # Perform the grid search for each model and print the best hyperparameters
    for model_name, model in models.items():
        print(colored(f"Tuning hyperparameters for {model_name}...", 'red', attrs=['bold']))
        
        clf = GridSearchCV(model, param_grid[model_name], scoring=scoring_metric, cv=cv)
        clf.fit(X_train, Y_train)

        print(f"Best hyperparameters: {clf.best_params_}")
        print(f"Training accuracy: {clf.best_score_:.4f}")
        validation_accuracy = accuracy_score(Y_test, clf.predict(X_test))
        print(f"Validation accuracy: {validation_accuracy:.4f}")

        # Predictions for training and testing data
        y_train_pred = clf.predict(X_train)
        y_test_pred = clf.predict(X_test)
        
        # Calculate performance metrics
        train_metrics = {
            'accuracy': accuracy_score(Y_train, y_train_pred),

        }
        
        test_metrics = {
            'accuracy': accuracy_score(Y_test, y_test_pred),
            'precision': precision_score(Y_test, y_test_pred, average='weighted'),
            'recall': recall_score(Y_test, y_test_pred, average='weighted'),

Bagging (Bootstrap Aggregation):

This technique involves training multiple instances of the same algorithm on different subsets of the data.

In[38]:
import matplotlib.pyplot as plt
import numpy as np
from sklearn.metrics import confusion_matrix, accuracy_score, f1_score
import seaborn as sns

def plot_confusion_matrix(y_true, y_pred, model_name):
    cm = confusion_matrix(y_true, y_pred)
    plt.figure(figsize=(6,4))
    sns.heatmap(cm, annot=True, fmt="d", cmap="Blues")
    plt.title(f'Confusion Matrix for {model_name}')
    plt.ylabel('Actual Label')
    plt.xlabel('Predicted Label')
    plt.show()

# Dummy lists for storing accuracies
train_accuracies = []
test_accuracies = []

# Function to plot training vs testing accuracy
def plot_training_vs_testing(train_accuracies, test_accuracies, model_name):
    epochs = np.arange(1, len(train_accuracies)+1)
    plt.figure(figsize=(8,6))
    plt.plot(epochs, train_accuracies, label='Training Accuracy')
    plt.plot(epochs, test_accuracies, label='Testing Accuracy')
    plt.title(f'Training vs Testing Accuracy for {model_name}')
    plt.xlabel('Iterations')
    plt.ylabel('Accuracy')
    plt.legend()
    plt.show()

from sklearn.model_selection import cross_val_score
from sklearn.ensemble import BaggingClassifier
from sklearn.tree import DecisionTreeClassifier

# Initialize the lists to store accuracies for plotting
train_accuracies = []
test_accuracies = []

# Example for Bagging with Cross-Validation
bagging_model = BaggingClassifier(estimator=DecisionTreeClassifier(), n_estimators=100, random_state=42)

# Perform cross-validation to get training and test accuracy scores
# Using 'cv=5' for 5-fold cross-validation
cv_scores = cross_val_score(bagging_model, X_train, Y_train, cv=5)

# Record the cross-validation scores (this gives testing accuracies for each fold)
test_accuracies = list(cv_scores)

# Since cross_val_score only provides test accuracy, you might approximate train accuracy with out-of-fold errors
bagging_model.fit(X_train, Y_train)
train_accuracy = bagging_model.score(X_train, Y_train)
train_accuracies = [train_accuracy] * len(test_accuracies)  # Replicate train accuracy for each fold (for plotting)

# Plot the accuracy graphs
plot_training_vs_testing(train_accuracies, test_accuracies, "Bagging")

# Make predictions on the test set and plot confusion matrix
bagging_predictions = bagging_model.predict(X_test)
plot_confusion_matrix(Y_test, bagging_predictions, "Bagging")
